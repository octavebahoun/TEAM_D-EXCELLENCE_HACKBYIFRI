🤖 IA de Suivi Personnalisé des Étudiants

Guide complet d'implémentation pour le développeur backend — de la base de données au message personnalisé généré par IA.

📅 Mars 2026 🏗️ Laravel + Python FastAPI 🤖 LangChain + OpenRouter ⚡ 8 étapes

🎯 Objectif de la Fonctionnalité

Cette fonctionnalité ajoute une intelligence artificielle de suivi personnalisé qui :

  • 📊 Récupère les notes de chaque étudiant en base de données
  • 🧮 Calcule les moyennes par matière et la moyenne générale
  • 🔍 Identifie automatiquement les matières faibles (en dessous du seuil)
  • 📈 Détecte la tendance (progression ou régression)
  • ✉️ Génère un message personnalisé, empathique et actionnable via un LLM
  • 🔔 Envoie ce message en notification / email à l'étudiant
💡
Philosophie : On n'entraîne pas un modèle ML maison. On construit un contexte structuré (notes, tendances, profil) et on l'envoie à un LLM via OpenRouter qui génère le message. C'est léger, maintenable et immédiatement opérationnel.

② Service Python — Calcul du Contexte

📁
Fichier à créer : python/app/services/student_analysis_service.py

Ce service est le cœur de la fonctionnalité. Il récupère les données depuis MySQL, calcule tout, puis appelle la chain LangChain.

python/app/services/student_analysis_service.pyPYTHON
import os
import pymysql
from dotenv import load_dotenv
from langchain_openai import ChatOpenAI
from langchain.prompts import PromptTemplate
from langchain_core.output_parsers import JsonOutputParser

load_dotenv()

# ──────────────────────────────────────────────
# 1. CONNEXION BASE DE DONNÉES
# ──────────────────────────────────────────────
def get_db_connection():
    return pymysql.connect(
        host=os.getenv("DB_HOST", "127.0.0.1"),
        port=int(os.getenv("DB_PORT", 3306)),
        user=os.getenv("DB_USER"),
        password=os.getenv("DB_PASSWORD"),
        database=os.getenv("DB_NAME"),
        cursorclass=pymysql.cursors.DictCursor
    )

# ──────────────────────────────────────────────
# 2. RÉCUPÉRATION DES NOTES
# ──────────────────────────────────────────────
def get_student_grades(student_id: int) -> dict:
    """
    Récupère les notes de l'étudiant groupées par matière.
    Retourne : { nom, filiere, objectif_moyenne, notes_par_matiere, absences }
    """
    conn = get_db_connection()
    try:
        with conn.cursor() as cursor:
            # Infos de base de l'étudiant
            cursor.execute("""
                SELECT u.name, u.email, p.filiere, p.niveau,
                       p.objectif_moyenne, p.absences
                FROM users u
                JOIN student_profiles p ON p.user_id = u.id
                WHERE u.id = %s
            """, (student_id,))
            student = cursor.fetchone()

            # Notes par matière (plusieurs évaluations possibles par matière)
            cursor.execute("""
                SELECT m.nom AS matiere, n.valeur, n.created_at
                FROM notes n
                JOIN matieres m ON m.id = n.matiere_id
                WHERE n.student_id = %s
                ORDER BY m.nom, n.created_at DESC
            """, (student_id,))
            notes = cursor.fetchall()

        return {"student": student, "notes": notes}
    finally:
        conn.close()

# ──────────────────────────────────────────────
# 3. CALCUL DES MOYENNES ET FAIBLESSES
# ──────────────────────────────────────────────
def build_student_context(student_id: int) -> dict:
    """
    Construit le contexte complet à envoyer au LLM.
    """
    data = get_student_grades(student_id)
    student = data["student"]
    notes = data["notes"]

    # Grouper les notes par matière
    matieres_dict = {}
    for note in notes:
        nom = note["matiere"]
        if nom not in matieres_dict:
            matieres_dict[nom] = []
        matieres_dict[nom].append(float(note["valeur"]))

    # Calculer les moyennes par matière
    matieres = []
    toutes_moyennes = []
    for nom, valeurs in matieres_dict.items():
        moyenne = round(sum(valeurs) / len(valeurs), 2)
        toutes_moyennes.append(moyenne)

        if moyenne < 8:
            statut = "critique"
        elif moyenne < 10:
            statut = "faible"
        elif moyenne < 13:
            statut = "moyen"
        else:
            statut = "fort"

        matieres.append({"nom": nom, "moyenne": moyenne, "statut": statut})

    moyenne_generale = round(sum(toutes_moyennes) / len(toutes_moyennes), 2) if toutes_moyennes else 0

    # Tendance : comparer les 3 dernières notes avec les 3 précédentes
    tendance = detect_trend(notes)

    return {
        "nom": student["name"],
        "filiere": f"{student['filiere']} {student.get('niveau', '')}".strip(),
        "moyenne_generale": moyenne_generale,
        "matieres": sorted(matieres, key=lambda x: x["moyenne"]),
        "tendance": tendance,
        "absences": student.get("absences", 0),
        "objectif_moyenne": float(student.get("objectif_moyenne", 12.0))
    }

# ──────────────────────────────────────────────
# 4. DÉTECTION DE TENDANCE
# ──────────────────────────────────────────────
def detect_trend(notes: list) -> str:
    """
    Compare les 3 notes les plus récentes aux 3 précédentes.
    """
    if len(notes) < 4:
        return "insuffisant"  # pas assez de données

    recentes = [float(n["valeur"]) for n in notes[:3]]
    precedentes = [float(n["valeur"]) for n in notes[3:6]]

    moy_rec = sum(recentes) / len(recentes)
    moy_prec = sum(precedentes) / len(precedentes)

    diff = moy_rec - moy_prec
    if diff > 1.5:
        return "en_hausse"
    elif diff < -1.5:
        return "en_baisse"
    else:
        return "stable"

# ──────────────────────────────────────────────
# 5. GÉNÉRATION DU MESSAGE VIA LANGCHAIN
# ──────────────────────────────────────────────
def generate_personalized_message(context: dict) -> dict:
    """
    Envoie le contexte structuré au LLM via LangChain + OpenRouter.
    Retourne un JSON structuré.
    """
    llm = ChatOpenAI(
        model=os.getenv("OPENROUTER_MODEL", "mistralai/mistral-7b-instruct"),
        openai_api_key=os.getenv("OPENROUTER_API_KEY"),
        openai_api_base=os.getenv("OPENROUTER_BASE_URL"),
        temperature=0.7,
    )

    template = PromptTemplate(
        input_variables=["nom", "filiere", "moyenne_generale", "matieres",
                         "tendance", "absences", "objectif_moyenne"],
        template="""
Tu es un tuteur académique bienveillant de la plateforme AcademiX.
Analyse les données suivantes d'un étudiant et génère un retour personnalisé.

DONNÉES ÉTUDIANT :
- Nom : {nom}
- Filière : {filiere}
- Moyenne générale : {moyenne_generale}/20
- Objectif de moyenne : {objectif_moyenne}/20
- Tendance : {tendance}
- Absences : {absences}
- Matières : {matieres}

CONSIGNES :
1. Rédige un message d'introduction chaleureux et personnalisé (3-4 phrases max)
2. Identifie les matières prioritaires à travailler (statut critique ou faible)
3. Donne 2-3 conseils concrets et actionnables
4. Termine sur une note positive en valorisant un point fort
5. Adapte le ton à la situation (encourageant si en hausse, mobilisateur si en baisse)

IMPORTANT : Réponds UNIQUEMENT en JSON valide avec cette structure exacte :
{{
  "message_principal": "...",
  "niveau_alerte": "info|warning|danger",
  "conseils": ["conseil1", "conseil2", "conseil3"],
  "matieres_prioritaires": ["matiere1", "matiere2"],
  "point_positif": "..."
}}
"""
    )

    chain = template | llm | JsonOutputParser()
    
    result = chain.invoke({
        "nom": context["nom"],
        "filiere": context["filiere"],
        "moyenne_generale": context["moyenne_generale"],
        "matieres": str(context["matieres"]),
        "tendance": context["tendance"],
        "absences": context["absences"],
        "objectif_moyenne": context["objectif_moyenne"]
    })

    return result

# ──────────────────────────────────────────────
# 6. POINT D'ENTRÉE PRINCIPAL
# ──────────────────────────────────────────────
def run_analysis(student_id: int) -> dict:
    """
    Fonction principale appelée par l'endpoint FastAPI.
    Retourne l'analyse complète.
    """
    context = build_student_context(student_id)
    message = generate_personalized_message(context)
    return {
        "student_id": student_id,
        "context": context,
        "analysis": message
    }

from fastapi import APIRouter, HTTPException
from pydantic import BaseModel
from app.services.student_analysis_service import run_analysis, build_student_context

router = APIRouter(prefix="/analysis", tags=["AI Analysis"])

# ── GET /analysis/{student_id} ──────────────────
@router.get("/{student_id}")
async def get_analysis(student_id: int):
    """
    Lance une analyse complète pour un étudiant.
    Retourne le contexte + le message IA généré.
    """
    try:
        result = run_analysis(student_id)
        return {"success": True, "data": result}
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

# ── GET /analysis/{student_id}/context ──────────
@router.get("/{student_id}/context")
async def get_context_only(student_id: int):
    """
    Retourne uniquement le contexte calculé (sans appeler le LLM).
    Utile pour debug / aperçu des données.
    """
    try:
        context = build_student_context(student_id)
        return {"success": True, "data": context}
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

StudentAnalysisController.phpPHP
<?php

namespace App\Http\Controllers;

use App\Models\User;
use App\Models\StudentAnalysis;
use App\Services\PythonAIService;
use Illuminate\Http\Request;

class StudentAnalysisController extends Controller
{
    public function __construct(protected PythonAIService $aiService) {}

    /**
     * Lancer une analyse IA pour un étudiant.
     * GET /api/v1/students/{id}/analysis
     */
    public function analyze(int $id)
    {
        $student = User::findOrFail($id);

        // Anti-spam : 1 analyse maximum toutes les 24h
        $lastAnalysis = StudentAnalysis::where('user_id', $id)
            ->where('created_at', '>=', now()->subHours(24))
            ->first();

        if ($lastAnalysis) {
            return response()->json([
                'success' => false,
                'message' => 'Une analyse a déjà été effectuée il y a moins de 24h.',
                'data'    => $lastAnalysis
            ], 429);
        }

        // Appel au service Python
        $result = $this->aiService->analyzeStudent($id);

        if (!$result['success']) {
            return response()->json([
                'success' => false,
                'message' => 'Erreur du service IA : ' . $result['error']
            ], 500);
        }

        $analysis = $result['data']['analysis'];
        $context  = $result['data']['context'];

        // Sauvegarder en base
        $record = StudentAnalysis::create([
            'user_id'              => $id,
            'moyenne_generale'     => $context['moyenne_generale'],
            'niveau_alerte'        => $analysis['niveau_alerte'],
            'message_principal'    => $analysis['message_principal'],
            'conseils'             => $analysis['conseils'],
            'matieres_prioritaires'=> $analysis['matieres_prioritaires'],
            'point_positif'        => $analysis['point_positif'] ?? null,
            'contexte_raw'         => $context,
        ]);

        return response()->json([
            'success' => true,
            'data'    => $record
        ]);
    }

    /**
     * Envoyer le message IA à l'étudiant par notification.
     * POST /api/v1/students/{id}/analysis/send
     */
    public function send(int $id)
    {
        $latestAnalysis = StudentAnalysis::where('user_id', $id)
            ->latest()
            ->firstOrFail();

        // TODO : envoyer via Email / Notification Laravel
        // $student->notify(new StudentAnalysisNotification($latestAnalysis));

        $latestAnalysis->update(['sent_at' => now()]);

        return response()->json([
            'success' => true,
            'message' => 'Message envoyé avec succès.',
            'sent_at' => $latestAnalysis->sent_at
        ]);
    }

    /**
     * Historique des analyses d'un étudiant.
     * GET /api/v1/students/{id}/analysis/history
     */
    public function history(int $id)
    {
        $analyses = StudentAnalysis::where('user_id', $id)
            ->orderBy('created_at', 'desc')
            ->paginate(10);

        return response()->json([
            'success' => true,
            'data'    => $analyses
        ]);
    }
}

⑥ Service HTTP Laravel → Python

📁
Fichier à créer : backend/laravel/app/Services/PythonAIService.php

Ce service est le client HTTP qui fait le pont entre Laravel et le service Python FastAPI.

app/Services/PythonAIService.phpPHP
<?php

namespace App\Services;

use Illuminate\Support\Facades\Http;
use Illuminate\Support\Facades\Log;

class PythonAIService
{
    private string $baseUrl;

    public function __construct()
    {
        $this->baseUrl = config('services.python_ai.url', 'http://localhost:5000');
    }

    /**
     * Lance l'analyse IA pour un étudiant.
     */
    public function analyzeStudent(int $studentId): array
    {
        try {
            $response = Http::timeout(30)
                ->get("{$this->baseUrl}/api/analysis/{$studentId}");

            if ($response->successful()) {
                return ['success' => true, 'data' => $response->json('data')];
            }

            Log::error("PythonAIService error", [
                'status' => $response->status(),
                'body'   => $response->body()
            ]);

            return ['success' => false, 'error' => 'Service IA indisponible'];

        } catch (\Exception $e) {
            Log::error("PythonAIService exception: " . $e->getMessage());
            return ['success' => false, 'error' => $e->getMessage()];
        }
    }
}

namespace App\Notifications;

use App\Models\StudentAnalysis;
use Illuminate\Notifications\Notification;
use Illuminate\Notifications\Messages\MailMessage;

class StudentAnalysisNotification extends Notification
{
    public function __construct(public StudentAnalysis $analysis) {}

    public function via($notifiable): array
    {
        return ['mail', 'database'];
    }

    public function toMail($notifiable): MailMessage
    {
        return (new MailMessage)
            ->subject('📊 Ton bilan académique personnalisé — AcademiX')
            ->greeting('Bonjour ' . $notifiable->name . ' !')
            ->line($this->analysis->message_principal)

            ->action('Voir mon tableau de bord', url('/dashboard'));

    {
        return [
            'type'              => 'student_analysis',
            'niveau_alerte'     => $this->analysis->niveau_alerte,
            'message_principal' => $this->analysis->message_principal,
            'matieres'          => $this->analysis->matieres_prioritaires,
        ];
    }
}
